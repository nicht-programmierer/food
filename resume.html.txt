        /* ================================================================
           星光粒子Canvas动画
           ================================================================ */
        (function initStars() {
            const canvas = document.getElementById('starCanvas');
            const ctx = canvas.getContext('2d');

            // 存储粒子信息
            const stars = [];
            const STAR_COUNT = 80; // 星光粒子数量

            // 粒子颜色：暖色系微光
            const starColors = [
                'rgba(255, 220, 180, 0.7)',
                'rgba(255, 200, 200, 0.6)',
                'rgba(220, 200, 240, 0.6)',
                'rgba(200, 230, 210, 0.6)',
                'rgba(255, 240, 200, 0.7)',
                'rgba(255, 255, 240, 0.8)',
            ];

            /**
             * 调整Canvas尺寸以匹配窗口
             */
            function resizeCanvas() {
                canvas.width = window.innerWidth;
                canvas.height = window.innerHeight;
            }

            /**
             * 初始化星光粒子
             */
            function createStars() {
                stars.length = 0;
                for (let i = 0; i < STAR_COUNT; i++) {
                    stars.push({
                        x: Math.random() * canvas.width,
                        y: Math.random() * canvas.height,
                        radius: Math.random() * 2.5 + 0.5,
                        color: starColors[Math.floor(Math.random() * starColors.length)],
                        // 闪烁速度与相位
                        speed: Math.random() * 0.02 + 0.005,
                        phase: Math.random() * Math.PI * 2,
                        // 缓慢漂移
                        driftX: (Math.random() - 0.5) * 0.3,
                        driftY: (Math.random() - 0.5) * 0.3,
                    });
                }
            }

            /**
             * 每帧绘制星光
             */
            function drawStars(timestamp) {
                ctx.clearRect(0, 0, canvas.width, canvas.height);

                stars.forEach(star => {
                    // 闪烁：通过正弦波控制透明度
                    const alpha = 0.3 + 0.7 * Math.abs(Math.sin(timestamp * star.speed + star.phase));

                    ctx.beginPath();
                    ctx.arc(star.x, star.y, star.radius, 0, Math.PI * 2);
                    ctx.fillStyle = star.color.replace(/[\d.]+\)$/, alpha.toFixed(2) + ')');
                    ctx.fill();

                    // 微光晕效果
                    ctx.beginPath();
                    ctx.arc(star.x, star.y, star.radius * 2.5, 0, Math.PI * 2);
                    ctx.fillStyle = star.color.replace(/[\d.]+\)$/, (alpha * 0.18).toFixed(3) + ')');
                    ctx.fill();

                    // 缓慢漂移
                    star.x += star.driftX;
                    star.y += star.driftY;

                    // 边界循环
                    if (star.x < -10) star.x = canvas.width + 10;
                    if (star.x > canvas.width + 10) star.x = -10;
                    if (star.y < -10) star.y = canvas.height + 10;
                    if (star.y > canvas.height + 10) star.y = -10;
                });

                requestAnimationFrame(drawStars);
            }

            // 初始化
            resizeCanvas();
            createStars();
            requestAnimationFrame(drawStars);

            // 窗口大小变化时重新调整
            window.addEventListener('resize', () => {
                resizeCanvas();
                createStars();
            });
        })();

        /* ================================================================
           滚动监听：切换body class以实现背景渐变微调
           ================================================================ */
        (function initScrollEffect() {
            let scrollTicking = false;

            function onScroll() {
                if (!scrollTicking) {
                    requestAnimationFrame(() => {
                        if (window.scrollY > 80) {
                            document.body.classList.add('scrolled');
                        } else {
                            document.body.classList.remove('scrolled');
                        }
                        scrollTicking = false;
                    });
                    scrollTicking = true;
                }
            }

            window.addEventListener('scroll', onScroll, { passive: true });
        })();

        (function initCardInteractions() {

                // 鼠标进入：角落图标微动
                card.addEventListener('mouseenter', () => {
                    const corners = card.querySelectorAll('.card-corner-decor');
                    corners.forEach((corner, i) => {

                        corner.style.opacity = '0.7';
                        corner.style.transform = 'scale(1.3)';
                        corner.style.transitionDelay = (i * 0.05) + 's';
                    });
                });

                // 鼠标离开：恢复
                card.addEventListener('mouseleave', () => {
                    const corners = card.querySelectorAll('.card-corner-decor');
                    corners.forEach(corner => {
                        corner.style.opacity = '0.35';
                        corner.style.transform = 'scale(1)';
                        corner.style.transitionDelay = '0s';
                    });
                });
            });
        })();
